ALG_casablanca_decisions = {
	ALG_casablanca_mission = {
		icon = GFX_west_sahara2

		visible = { has_country_flag = algeria_casablanca_unlocked }

		available = {
			custom_trigger_tooltip = {
				tooltip = ALG_casablanca_mission_tt
				check_variable = { algeria_casablanca_result = 0 }
				NOT = { check_variable = { algeria_casablanca_approach = 0 } }
			}
		}

		days_mission_timeout = 180
		cancel_if_not_visible = yes

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision ALG_casablanca_mission"
			# if approach = 1 (great powers) -> best odds
			if = {
				limit = { check_variable = { variable = algeria_casablanca_approach value = 1 compare = greater_than_or_equals } }
				random_list = {
					60 = { set_variable = { algeria_casablanca_result = 2 } } # high success
					30 = { set_variable = { algeria_casablanca_result = 1 } } # medium
					10 = { set_variable = { algeria_casablanca_result = 0 } } # low
				}
			}

			# approach = 3 (mixed) -> balanced odds
			if = {
				limit = { check_variable = { variable = algeria_casablanca_approach value = 3 compare = greater_than_or_equals } }
				random_list = {
					40 = { set_variable = { algeria_casablanca_result = 2 } }
					40 = { set_variable = { algeria_casablanca_result = 1 } }
					20 = { set_variable = { algeria_casablanca_result = 0 } }
				}
			}

			# approach = 2 (regional) -> worst odds
			if = {
				limit = { check_variable = { variable = algeria_casablanca_approach value = 2 compare = greater_than_or_equals } }
				random_list = {
					10 = { set_variable = { algeria_casablanca_result = 2 } }
					30 = { set_variable = { algeria_casablanca_result = 1 } }
					60 = { set_variable = { algeria_casablanca_result = 0 } }
				}
			}

			# fallback (no approach set) - treat as regional
			if = {
				limit = {
					NOT = {
						OR = {
							check_variable = { variable = algeria_casablanca_approach value = 1 compare = greater_than_or_equals }
							check_variable = { variable = algeria_casablanca_approach value = 2 compare = greater_than_or_equals }
							check_variable = { variable = algeria_casablanca_approach value = 3 compare = greater_than_or_equals }
						}
					}
				}
				random_list = {
					10 = { set_variable = { algeria_casablanca_result = 2 } }
					30 = { set_variable = { algeria_casablanca_result = 1 } }
					60 = { set_variable = { algeria_casablanca_result = 0 } }
				}
			}

			# Rewards: PP & stability only
			if = {
				limit = { check_variable = { variable = algeria_casablanca_result value = 2 compare = greater_than_or_equals } }
				add_political_power = 100
				add_stability = 0.03
				hidden_effect = { news_event = { id = AlgeriaFocusNews.AU_result_high hours = 6 } }
			}
			if = {
				limit = { check_variable = { variable = algeria_casablanca_result value = 1 compare = greater_than_or_equals } }
