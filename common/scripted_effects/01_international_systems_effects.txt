un_recog_setup_check = {
	if = {
		limit = {
			OR = {
				has_idea = Lacks_International_Recognition
				has_idea = Non_State_Actor
			}
		}
		if = {
			limit = {
				NOT = {
					is_in_array = { global.Unrecognised_States = THIS }
					THIS = { has_idea = Lacks_International_Recognition }
				}
			}
			add_to_array = { global.Unrecognised_States = THIS }
		} else = {
			if = {
				limit = { is_in_array = { global.Unrecognised_States = THIS } }
				remove_from_array = { global.Unrecognised_States = THIS }
			}
			add_to_array = {
				array = global.Unrecognised_States
				value = THIS
				index = 0
			}
		}
	}else = {
		add_to_array = { global.UN_general_assembly = THIS }
		add_dynamic_modifier = { modifier = united_nations_member }
	}
}

update_sc_vote = {
	if = {
		limit = {
			OR = {
				has_global_flag = current_ongoing_sc_vote
				has_global_flag = un_sc_vote_cooldown
				check_variable = { global.sc_vote_type_tracker^num > 0 }
			}
		}
		add_to_array = { global.sc_vote_type_tracker = sc_new_vote_type }
		add_to_array = { global.sc_vote_nation_tracker = sc_new_vote_nation }
		custom_effect_tooltip = add_vote_to_queue_tt
	} else = {
		set_variable = { global.current_sc_vote_type = sc_new_vote_type }
		set_variable = { global.current_sc_vote_nation = sc_new_vote_nation }
		if = {
			limit = {
				check_variable = { global.current_sc_vote_type = 5 }
			}
			var:global.current_sc_vote_nation = {
				hidden_effect = {
					country_event = recognition.10
				}
			}
		} else = {
			var:global.current_sc_vote_nation = {
				hidden_effect = {
					country_event = UNSC.100
				}
			}
		}
		custom_effect_tooltip = start_new_vote_tt
	}
}

		index = index_track
	}

	remove_from_array = {
		array = global.ga_vote_type_tracker
		index = index_track
	}

	for_each_loop = {
		array = global.ga_vote_type_tracker
		value = v
		index = i
		break = b

		if = {
			limit = {
				NOT = {
					check_variable = { v = 6 }
					check_variable = { v = 7 }
				}
			}
			set_temp_variable = { move_index = i }
			set_temp_variable = { b = 1 }
		}
	}

	add_to_array = {
		array = global.ga_vote_nation_tracker
		value = nation_hold
		index = move_index
	}

	add_to_array = {
		array = global.ga_vote_type_tracker
		value = type_hold
		index = move_index
	}
}

start_unsc_next_term_voting = {
	set_global_flag = electing_new_unsc_members
	set_variable = { global.electing_new_members_tracker = 0 } #use this to clear the global flag when the decision finishes

	set_temp_variable = { even_year_checker = global.year }
	modulo_temp_variable = { even_year_checker = 2 }

	if = {
		limit = {
			check_variable = { even_year_checker = 0 }
		}
		set_temp_variable = { aa_seat_tracker = 2 }
		set_temp_variable = { euro_others_seat_tracker = 2 }
		set_temp_variable = { south_america_seat_tracker = 1 }
	} else = {
		set_temp_variable = { aa_seat_tracker = 3 }
		set_temp_variable = { euro_others_seat_tracker = 1 }
		set_temp_variable = { south_america_seat_tracker = 1 }
	}

	for_loop_effect = {
		end = aa_seat_tracker
		if = {
			limit = {
				check_variable = { array_shift < 1 }
			}
			random_scope_in_array = {
				array = global.asian_nations
				limit = {
					NOT = { is_in_array = { global.security_council_members = THIS } }
					NOT = { has_idea = Lacks_International_Recognition has_idea = Non_State_Actor }
				}
				add_to_array = { global.new_council_members = THIS }
			}
		} else = {
			if = {
				limit = {
					check_variable = { array_shift < 2 }
				}
				random_scope_in_array = {
					array = global.sub_saharan_nations
					limit = {
						NOT = { is_in_array = { global.security_council_members = THIS } }
						NOT = { has_idea = Lacks_International_Recognition has_idea = Non_State_Actor }
					}
					add_to_array = { global.new_council_members = THIS }
				}
			} else = {
				random_scope_in_array = {
					array = global.middle_eastern_nations
					limit = {
						NOT = { is_in_array = { global.security_council_members = THIS } }
						NOT = { has_idea = Lacks_International_Recognition has_idea = Non_State_Actor }
					}
					add_to_array = { global.new_council_members = THIS }
				}
			}
		}
		add_to_temp_variable = { array_shift = 1 }
	}

	for_loop_effect = {
		end = euro_others_seat_tracker
		random_scope_in_array = {
			array = global.sc_updates_euro_america
			limit = {
				NOT = { is_in_array = { global.security_council_members = THIS } }
				NOT = { has_idea = Lacks_International_Recognition has_idea = Non_State_Actor }
			}
			add_to_array = { global.new_council_members = THIS }
		}
	}

	for_loop_effect = {
		end = south_america_seat_tracker
		random_scope_in_array = {
			array = global.american_nations
			limit = {
				NOT = { is_in_array = { global.security_council_members = THIS } }
				NOT = { has_idea = Lacks_International_Recognition has_idea = Non_State_Actor }
				capital_scope = {
					is_on_continent = south_america
				}
			}
			add_to_array = { global.new_council_members = THIS }
		}
	}
	#Fires in june, should be faster voting efforts (cycle them faster, bypass the queueing system)
	set_temp_variable = { ga_new_vote_type = 6 }
	set_temp_variable = { ga_new_vote_nation = global.new_council_members^0 }
	update_ga_vote = yes
}

update_security_council_peeps = {
	set_temp_variable = { set = 5 }
	set_temp_variable = { set_index = 0 }

	#Overwrite the current leaving members with held members
	#This pushes them forward in the array, so next year they leave
	for_loop_effect = {
		end = 5
		set_variable = { global.security_council_members^set_index = global.security_council_members^set }
		add_to_temp_variable = { set = 1 }
		add_to_temp_variable = { set_index = 1 }
	}

	#Add the new members here
	set_temp_variable = { set = 0 }
	set_temp_variable = { set_index = 5 }
	for_loop_effect = {
		end = 5
		set_variable = { global.security_council_members^set_index = global.confirmed_new_council_members^set }
		add_to_temp_variable = { set = 1 }
		add_to_temp_variable = { set_index = 1 }
	}

	clear_array = global.confirmed_new_council_members

	missile_ui_update = yes
}

#This makes it so that receiving money reduces your benefits from the main UN modifier
#While spending a higher percentage will increase your benefits
calculate_final_un_benefits = {
	#add up everything we are receiving aid wise
	set_temp_variable = { current_receiving_budget_tracker = un_esco_political_power_var }
	add_to_temp_variable = { current_receiving_budget_tracker = un_esco_research_speed_var }
	add_to_temp_variable = { current_receiving_budget_tracker = un_ido_construction_speed_var }

	set_temp_variable = { current_factory_lease_money = un_ido_current_building_count }
	multiply_temp_variable = { current_factory_lease_money = 0.14 }

	add_to_temp_variable = { current_receiving_budget_tracker = current_factory_lease_money }
	add_to_temp_variable = { current_receiving_budget_tracker = un_cdf_productivity_var }

	set_temp_variable = { current_const_cost_bonus = un_cdf_construction_cost_var }

	add_to_temp_variable = { current_receiving_budget_tracker = current_const_cost_bonus }

	#Divide here to see how much % of the money we are getting
	#then flip it to negative; we need to add this back into the nations that are getting benefits
	divide_temp_variable = { current_receiving_budget_tracker = global.full_aid_budget_calculator }
	multiply_temp_variable = { current_receiving_budget_tracker = -1 }
	multiply_temp_variable = { current_receiving_budget_tracker = 2 }

	#if you receive aid, you begin to lose some of the gains from giving aid
	set_temp_variable = { reducer = 0 }
	if = {
		limit = {
			has_country_flag = is_getting_ido_subsidies
		}
		add_to_temp_variable = { reducer = 0.2 }
	}
	if = {
		limit = {
			has_country_flag = is_getting_cdf_subsidies
		}
		add_to_temp_variable = { reducer = 0.2 }
	}
	if = {
		limit = {
			has_country_flag = is_getting_esco_subsidies
		}
		add_to_temp_variable = { reducer = 0.2 }
	}

	set_temp_variable = { contribution_multiplier = 1 }
	subtract_from_temp_variable = { contribution_multiplier = reducer }

	set_temp_variable = { aid_program_adder = united_nations_aid_program_contribution_percent }
	multiply_temp_variable = { aid_program_adder = contribution_multiplier }

	set_variable = { united_nations_political_power_var = aid_program_adder }
	multiply_variable = { united_nations_political_power_var = 0.5 }

	set_variable = { united_nations_influence_gain_var = united_nations_contribution_percent }
	add_to_variable = { united_nations_influence_gain_var = aid_program_adder }
	add_to_variable = { united_nations_influence_gain_var = current_receiving_budget_tracker }

	set_variable = { united_nations_influence_defense_var = united_nations_contribution_percent }
	add_to_variable = { united_nations_influence_defense_var = aid_program_adder }
	add_to_variable = { united_nations_influence_defense_var = current_receiving_budget_tracker }

	#Investments get calculated as the average of contributions
	set_temp_variable = { investment_adder = united_nations_contribution_percent }
	add_to_temp_variable = { investment_adder = aid_program_adder }
	multiply_temp_variable = { investment_adder = -0.5 }

	set_variable = { united_nations_investment_duration_var = investment_adder }
	set_variable = { united_nations_investment_cost_var = investment_adder }
}

check_for_subsidy_expiration = {
	set_temp_variable = { research_check = modifier@research_speed_factor }
	subtract_from_temp_variable = { research_check = un_esco_research_speed_var }
	if = {
		limit = {
			has_country_flag = is_getting_esco_subsidies
			OR = {
				check_variable = { research_check > global.target_research_speed_mod }
				check_variable = { political_power_daily > 7 }
			}
		}
		clr_country_flag = is_getting_esco_subsidies
		remove_from_array = { global.unesco_recipients = THIS }
		country_event = { id = UN.300 }
	}
	set_temp_variable = { construction_check = modifier@production_speed_buildings_factor }
	subtract_from_temp_variable = { construction_check = un_ido_construction_speed_var }
	if = {
		limit = {
			has_country_flag = is_getting_ido_subsidies
			OR = {
				check_variable = { construction_check > global.target_const_speed_mod }
				check_variable = { industrial_complex_total > global.target_civ_count }
			}
		}
		clr_country_flag = is_getting_ido_subsidies
		remove_from_array = { global.unido_recipients = THIS }
		country_event = { id = UN.301 }
	}
	if = {
		limit = {
			has_country_flag = is_getting_cdf_subsidies
			OR = {
				check_variable = { gdp_total > 1000 }
				check_variable = { overall_productivity > global.world_productivity_target }
			}
		}
		clr_country_flag = is_getting_cdf_subsidies
		remove_from_array = { global.uncdf_recipients = THIS }
		country_event = { id = UN.302 }
	}
	if = {
		limit = {
			has_dynamic_modifier = { modifier = un_aid_subsidies_modifier }
			NOT = {
				has_country_flag = is_getting_esco_subsidies
				has_country_flag = is_getting_ido_subsidies
				has_country_flag = is_getting_cdf_subsidies
			}
		}
		remove_dynamic_modifier = { modifier = un_aid_subsidies_modifier }
	}
}
