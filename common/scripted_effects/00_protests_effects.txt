apply_protest_effects = {
	# monthly cooldown of tensions
	apply_passive_protest_drift = yes
	apply_passive_radicalisation_drift = yes

	set_temp_variable = { protest_drift = 0 }
	calculate_protest_drift = yes

	add_to_variable = { protest_strength = protest_drift }
}

apply_passive_protest_drift = {
	random = {
		chance = 50
		subtract_from_variable = { protest_strength = 2 }
	}

	if = { limit = { NOT = { has_variable = protest_strength } } set_variable = { protest_strength = 0 } }

	clamp_variable = { var = protest_strength min = 0 max = 100 }
}

apply_passive_radicalisation_drift = {
	random = {
		chance = 10
		subtract_from_variable = { protest_radicalisation = 1 }
	}

	if = { limit = { NOT = { has_variable = protest_radicalisation } } set_variable = { protest_radicalisation = 0 } }

	clamp_variable = { var = protest_radicalisation min = 0 max = 10 }
}

calculate_protest_drift = {
	if = { limit = { has_fuel < 1 }
		add_to_temp_variable = { protest_drift = 1 }
	}

	set_temp_variable = { buf = expected_welfare_spending }
	subtract_from_temp_variable = { buf = social_budget }
	add_to_temp_variable = { protest_drift = buf }

	set_temp_variable = { buf = expected_health_spending }
	subtract_from_temp_variable = { buf = health_budget }
	add_to_temp_variable = { protest_drift = buf }

	set_temp_variable = { buf = expected_edu_spending }
	subtract_from_temp_variable = { buf = education_budget }
	add_to_temp_variable = { protest_drift = buf }

	set_temp_variable = { buf = expected_police_spending }
	subtract_from_temp_variable = { buf = crime_fighting }
	add_to_temp_variable = { protest_drift = buf }

	set_temp_variable = { buf = expected_adm_spending }
	subtract_from_temp_variable = { buf = bureaucracy }
	add_to_temp_variable = { protest_drift = buf }

	set_temp_variable = { buf = expected_military_sp }
	subtract_from_temp_variable = { buf = Military_Spending }
	add_to_temp_variable = { protest_drift = buf }

	multiply_temp_variable = { protest_drift = 0.5 }
}

calculate_unrest_growth = {
	#Grab our old protest strength
	set_temp_variable = { start_protest_str = protest_strength }
	#Calculate based off economics how much the protests should drift
	calculate_protest_drift = yes
	add_to_variable = { protest_strength = protest_drift }

	#low stability in elected government = unrest
	if = {
		limit = {
			has_stability < 0.2
			has_elections = yes
		}
		set_temp_variable = { stab_hold = stability }
		set_temp_variable = { protest_add = 0.2 }
		subtract_from_temp_variable = { protest_add = stab_hold }
		add_to_variable = { protest_strength = protest_add }
	}
	#High interest = unrest
	if = {
		limit = { check_variable = { interest_rate > 8.9 } }
		set_temp_variable = { rate_hold = interest_rate }
		subtract_from_temp_variable = { rate_hold = 8.9 }
		add_to_variable = { protest_strength = protest_drift }
	}

	if = {
		limit = { has_idea = recession }
		randomize_temp_variable = {
			var = protest_drift
			distribution = poisson
			min = 0.5
			lambda = 3
		}
		add_to_variable = { protest_strength = protest_drift }
	}

	if = {
		limit = { has_idea = depression }
		randomize_temp_variable = {
			var = protest_drift
			distribution = poisson
			min = 0.5
			lambda = 4
		}
		add_to_variable = { protest_strength = protest_drift }
	}

	#Long wars = unrest
	if = {
		limit = { check_variable = { months_at_war > 36 } }
		set_temp_variable = { months_hold = months_at_war }
		divide_temp_variable = { months_hold = 24 }
		add_to_variable = { protest_strength = months_hold }
	}

	#If goverment popularity is less than 30%, check for needing elections or corruption
	if = {
		limit = {
			check_variable = { government_coalition_strength < 0.3 }
			OR = {
				has_elections = no
				OR = {
					has_idea = paralyzing_corruption
					has_idea = crippling_corruption
					has_idea = rampant_corruption
					has_idea = unrestrained_corruption
					has_idea = systematic_corruption
					has_idea = widespread_corruption
				}
			}
		}

		set_temp_variable = { protest_drift = 0.3 }
		set_temp_variable = { gov_pop_hold = government_coalition_strength }
		subtract_from_temp_variable = { protest_drift = government_coalition_strength }
		add_to_variable = { protest_strength = protest_drift }
	}

	if = {
		limit = {
			OR = {
				has_idea = CT_internet_monitoring
				has_idea = CT_protesting_arrests
			}
		}

		randomize_temp_variable = {
			var = protest_drift
			distribution = poisson
			min = 0.5
			lambda = 5
		}
		add_to_variable = { protest_strength = protest_drift }
	}

	if = {
		limit = {
			check_variable = { protest_strength > 0 }
			check_variable = { start_protest_str = protest_strength }
		}

		randomize_temp_variable = {
			var = protest_reduction
			distribution = poisson
			min = 0.5
			lambda = 3
		}

		subtract_from_variable = { protest_strength = protest_reduction }
	}

	round_variable = protest_strength

	clamp_variable = { var = protest_strength min = 0 max = 100 }
	check_protest_trigger = yes
}

check_protest_trigger = {
	if = {
		limit = { NOT = { has_country_flag = MD_protests_began } check_variable = { protest_strength > 49 } }

		set_temp_variable = { protest_roll = protest_strength }
		round_temp_variable = protest_roll
		subtract_from_temp_variable = { protest_roll = 49 }

		clamp_variable = { var = protest_roll max = 100 }

		random = {
			chance = var:protest_roll
			country_event = { id = protests.200 days = 1 }
			activate_mission = MD_incoming_protest_movement
		}
	}
}
