SWE_modify_royal_acceptance = {
	add_to_variable = { SWE_royal_acceptance_status = modify_royal_acceptance }
	clamp_variable = { var = SWE_royal_acceptance_status min = 1 max = 100 }
	custom_effect_tooltip = SWE_modify_royal_acceptance_tt
}
SWE_royal_acceptance_resolve = {
	if = {
		limit = {
			check_variable = { SWE_royal_acceptance_status < 25 }
		}
		random = {
			chance = 20
			country_event = { id = Sweden_royal.8 days = 1 }
		}
	}
}
