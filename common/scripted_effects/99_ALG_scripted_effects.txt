ALG_increase_desertification_rate = {
	if = {
		limit = {
			check_variable = { sahara_retreats > 4 }
		}
		custom_effect_tooltip = ALG_increase_yearly_bad_desertification_5
		add_to_variable = { sahara_expands = 5 }
		custom_effect_tooltip = ALG_reduce_yearly_good_desertification_5
		subtract_from_variable = { sahara_retreats = 5 }
	}
	else_if = {
		limit = {
			check_variable = { sahara_steady > 4 }
		}
		custom_effect_tooltip = ALG_increase_yearly_bad_desertification_5
		add_to_variable = { sahara_expands = 5 }
		custom_effect_tooltip = ALG_reduce_yearly_neutral_desertification_5
		subtract_from_variable = { sahara_steady = 5 }
	}
	else = {
		add_political_power = -200
	}
}

ALG_update_anti_poaching = {
	# From internal security level
	if = {
		limit = { has_idea = police_01 }
		set_variable = { chance_to_kill_poachers = 15 }
	}
	else_if = {
		limit = { has_idea = police_02 }
		set_variable = { chance_to_kill_poachers = 30 }
	}
	else_if = {
		limit = { has_idea = police_03 }
		set_variable = { chance_to_kill_poachers = 45 }
	}
	else_if = {
		limit = { has_idea = police_04 }
		set_variable = { chance_to_kill_poachers = 60 }
	}
	else_if = {
		limit = { has_idea = police_05 }
		set_variable = { chance_to_kill_poachers = 75 }
	}

	# From hiring extra security
	if = {
		limit = { has_decision = ALG_anti_poaching_extra_security }
		add_to_variable = { chance_to_kill_poachers = 10 }
	}

	ALG_update_cabinet_idea = yes

ALG_update_cabinet_idea = {
	# Remove all cabinet ideas first
	if = {
		limit = { has_idea = ALG_cabinet_loyal }
		remove_ideas = ALG_cabinet_loyal
	}
	if = {
		limit = { has_idea = ALG_cabinet_competent }
		remove_ideas = ALG_cabinet_competent
	}
	if = {
		limit = { has_idea = ALG_cabinet_disloyal }
		remove_ideas = ALG_cabinet_disloyal
	}
	if = {
		limit = { has_idea = ALG_cabinet_incompetent }
		remove_ideas = ALG_cabinet_incompetent
	}
	if = {
		limit = { has_idea = ALG_cabinet_deep_state_dominated }
		remove_ideas = ALG_cabinet_deep_state_dominated
	}
	if = {
		limit = { has_idea = ALG_cabinet_balanced }
		remove_ideas = ALG_cabinet_balanced
	}
	if = {
		limit = { has_idea = ALG_cabinet_reformist }
		remove_ideas = ALG_cabinet_reformist
	}

	# Add appropriate idea based on cabinet composition
	# Priority: Disloyal > Deep State Dominated > Incompetent > Loyal > Competent > Reformist > Balanced

	if = {
		limit = {
			check_variable = { ALG_cabinet_loyalty < 40 }
		}
		add_ideas = ALG_cabinet_disloyal
	}
	else_if = {
		limit = {
			check_variable = { ALG_cabinet_deep_state_influence > 70 }
		}
		add_ideas = ALG_cabinet_deep_state_dominated
	}
	else_if = {
		limit = {
			check_variable = { ALG_cabinet_competence < 40 }
		}
		add_ideas = ALG_cabinet_incompetent
	}
	else_if = {
		limit = {
			check_variable = { ALG_cabinet_loyalty > 75 }
		}
		add_ideas = ALG_cabinet_loyal
	}
	else_if = {
		limit = {
			check_variable = { ALG_cabinet_competence > 70 }
		}
		add_ideas = ALG_cabinet_competent
	}
	else_if = {
		limit = {
			check_variable = { ALG_cabinet_deep_state_influence < 30 }
		}
		add_ideas = ALG_cabinet_reformist
	}
	else = {
		add_ideas = ALG_cabinet_balanced
	}
}
