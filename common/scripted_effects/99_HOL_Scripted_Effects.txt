HOL_adjust_referendum_outcome = {
	if = {
		limit = { BEL = { HOL = { is_top3_influencer = yes } } }
		add_to_variable = { var = HOL_belgium_referendum_outcome value = 20 }
	}
	if = {
		limit = { BEL = { FRA = { is_top3_influencer = yes } } }
		add_to_variable = { var = HOL_belgium_referendum_outcome value = 15 }
	}
	if = {
		limit = { BEL = { GER = { is_top5_influencer = yes } } }
		add_to_variable = { var = HOL_belgium_referendum_outcome value = 10 }
	}
	if = {
		limit = { BEL = { has_stability < 50 } }
		add_to_variable = { var = HOL_belgium_referendum_outcome value = 20 }
	}
}
