assign_power_ranking_value = {
	if = { limit = { check_variable = { percentage_of_global_factories < 0.002 } }
		set_variable = { power_ranking_name_var = 1 }
	}
	else = {
		if = { limit = { check_variable = { percentage_of_global_factories > 0.15 } }
			set_variable = { power_ranking_name_var = 2 }
		}
		else_if = { limit = { check_variable = { percentage_of_global_factories > 0.065 } }
			set_variable = { power_ranking_name_var = 3 }
		}
		else_if = { limit = { check_variable = { percentage_of_global_factories > 0.025 } }
			set_variable = { power_ranking_name_var = 4 }
		} else_if = { limit = { always = yes }
			set_variable = { power_ranking_name_var = 5 }
		}
	}
}

sort_and_assign_power_rankings = {
	# Sort each regional array (descending by power_ranking_contribution)
	power_ranking_selection_sort_european = yes
	power_ranking_selection_sort_sub_saharan = yes
	power_ranking_selection_sort_middle_east = yes

	determine_regional_powers = yes
}

power_ranking_selection_sort_european = {
	for_each_loop = {
		array = global.european_nations
		value = v
		var:v = {
			add_to_temp_array = { pr_sort_values = power_ranking_contribution }
			add_to_temp_array = { pr_sort_nations = THIS }
		}
	}

	clear_array = global.european_nations
	set_temp_variable = { pr_sort_end = pr_sort_values^num }
	for_loop_effect = {
		end = pr_sort_end

		find_highest_in_array = {
			array = pr_sort_values
			index = pr_sort_idx
		}

		set_temp_variable = { pr_sort_nation = pr_sort_nations^pr_sort_idx }
		add_to_array = { global.european_nations = pr_sort_nation }

		remove_from_temp_array = { array = pr_sort_nations index = pr_sort_idx }
		remove_from_temp_array = { array = pr_sort_values index = pr_sort_idx }
	}
}

# Shared: Calculate percentage and assign power ranking ideas to every country
assign_power_ranking_ideas = {
	set_variable = { global.cur_nation_idea_pull = power_ranking_name_var }
	meta_effect = {
		text = {
			if = {
				limit = { NOT = { has_idea = [IDEA] } }
				remove_power_ideas = yes
				add_ideas = [IDEA]
			}
		}
		IDEA = "[THIS.PRGetIdeaName]"
	}
}
