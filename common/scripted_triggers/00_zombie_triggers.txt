ZOM_spawn_delay_passed = {
	OR = {
		has_game_rule = { rule = zombie_spawn_delay option = NONE }
		AND = {
			has_game_rule = { rule = zombie_spawn_delay option = ONE_YEAR }
			date > 2001.1.1
		}
		AND = {
			has_game_rule = { rule = zombie_spawn_delay option = TWO_YEARS }
			date > 2002.1.1
		}
		AND = {
			has_game_rule = { rule = zombie_spawn_delay option = THREE_YEARS }
			date > 2003.1.1
		}
	}
}

ZOM_coalition_can_form = {
	has_game_rule = { rule = zombie_coalition_enabled option = YES }
	ZOM_spawn_delay_passed = yes
}

ZOM_difficulty_at_least_hard = {
	OR = {
		has_game_rule = { rule = zombie_difficulty option = HARD }
		has_game_rule = { rule = zombie_difficulty option = EXTREME }
		has_game_rule = { rule = zombie_difficulty option = INSANE }
	}
}

ZOM_difficulty_at_least_extreme = {
	OR = {
		has_game_rule = { rule = zombie_difficulty option = EXTREME }
		has_game_rule = { rule = zombie_difficulty option = INSANE }
	}
}

ZOM_difficulty_insane = {
	has_game_rule = { rule = zombie_difficulty option = INSANE }
}
