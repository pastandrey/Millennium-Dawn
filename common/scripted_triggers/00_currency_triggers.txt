# Currency System — Scripted Triggers

is_reserve_currency_issuer = {
	OR = {
		has_country_flag = is_usd_issuer
		has_country_flag = is_eur_issuer
		has_country_flag = is_cny_issuer
		has_country_flag = is_rub_issuer
		has_country_flag = is_gbp_issuer
		has_country_flag = is_jpy_issuer
		has_country_flag = is_chf_issuer
	}
}

has_strong_currency = {
	check_variable = { currency_strength > 1.1 }
}

has_weak_currency = {
	check_variable = { currency_strength < 0.7 }
}

has_currency_crisis = {
	has_country_flag = in_currency_crisis
}
