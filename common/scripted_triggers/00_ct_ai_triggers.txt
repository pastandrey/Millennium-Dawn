# Classifies nations as a major power or regional power in the system
is_ct_major_power = {
	OR = {
		has_idea = super_power
		has_idea = great_power
	}
}

is_ct_regional_power = {
	OR = {
		has_idea = large_power
		has_idea = regional_power
	}
}

# Checks our region, based on the number and the flag you have
ct_ai_same_region_as_org = {
	OR = {
		AND = {
			check_variable = { ct_check_region = 0 }
			has_country_flag = european_nation_flag
		}
		AND = {
			check_variable = { ct_check_region = 1 }
			has_country_flag = africa_nation_flag
		}
		AND = {
			check_variable = { ct_check_region = 2 }
			has_country_flag = middle_east_flag
		}
		AND = {
			check_variable = { ct_check_region = 3 }
			has_country_flag = asian_nation_flag
		}
		AND = {
			check_variable = { ct_check_region = 4 }
			NOT = {
				OR = {
					has_country_flag = european_nation_flag
					has_country_flag = africa_nation_flag
					has_country_flag = middle_east_flag
					has_country_flag = asian_nation_flag
				}
			}
		}
	}
}
