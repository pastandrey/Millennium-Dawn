ITA_supports_afghanistan = {
	allowed = { tag = ITA }
	enable = {
		NOT = { has_war_with = AFG }
		AFG = { NOT = { has_government = fascism } }
		OR = {
			AND = {
				USA = { has_country_flag = Operation_enduring_freedom }
				OR = {
					has_country_flag = partially_supported_us_nine_eleven
					has_country_flag = fully_supported_us_nine_eleven
				}
			}
			date > 2002.6.1
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = befriend id = "AFG" value = 50 }
	ai_strategy = { type = support id = "AFG" value = 150 }
	ai_strategy = { type = send_volunteers_desire id = "AFG" value = 50 }
}
