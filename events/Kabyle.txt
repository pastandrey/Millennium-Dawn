	desc = KBY.10.d

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		has_country_flag = KBY_independence_recognized
	}

	option = {
		name = KBY.10.a

			factor = 100
		}

				target = TUR
				modifier = new_embassies
			}
		}
		add_named_threat = {
			threat = 5
			name = KBY_diplomatic_push
		}
	}
}
