		}

			name = "Milisia"
			regiments = {
				Militia_Bat = { x = 0 y = 0 }
			}
			priority = 2
		}

			limit = {
				has_country_flag = MEF_coup

					has_government = nationalist

				}
			}
			random_country_with_original_tag = {
				original_tag_to_check = SOL
				limit = {
					has_government = neutrality

						has_country_flag = SOL_CW_Active

				create_country_leader = {
					name = "Andrew Te'e"
					picture = "Portrait_Andrew_Tee.dds"
					ideology = Nat_Autocracy
					traits = {
						nationalist_Nat_Autocracy
						ruthless
					}
				}
			}
		}
		else = {
			random_country_with_original_tag = {
				original_tag_to_check = SOL
				limit = {
					has_government = neutrality
					NOT = {
						has_country_flag = SOL_CW_Active
					}
				}

				set_country_flag = SOL_IFM_id
				set_cosmetic_tag = SOL_IFM

				retire_country_leader = yes
				set_temp_variable = { rul_party_temp = 13 }
				change_ruling_party_effect = yes

				create_country_leader = {
					name = "Harold Keke"
					picture = "harold_keke.dds"
					ideology = Neutral_Autocracy
					traits = {
						neutrality_Neutral_Autocracy
					}
				}

				capital_scope = {
					create_unit = {
						division = "name = \"Armed grup blong Isatabu\" division_template = \"Ol faetman\"start_experience_factor = 0.1 start_equipment_factor = 0.8"
						owner = ROOT
						allow_spawning_on_enemy_provs = yes
					}
				}
			}

			clr_country_flag = SOL_CW_Greenlight

			#Honiara
			set_province_controller = 16318
