#Profits from Asteroid Mining
country_event = {
	id = ENG_military.005
	title = ENG_military.005.t
	desc = ENG_military.005.d
	picture = GFX_asteroid_in_orbit

	is_triggered_only = yes

	option = { #Excellent!
		name = ENG_military.005.a
		log = "[GetDateText]: [This.GetName]: ENG_military.005.a executed"
		if = {
			limit = {
				check_variable = {
					var = global.asteroid_mineral_count
					value = asteroid_mining_rate
					compare = greater_than_or_equals
				}
			}
			custom_effect_tooltip = asteroid_mining_mined_tt
			add_to_variable = { treasury = asteroid_mining_rate }
			subtract_from_variable = { global.asteroid_mineral_count = asteroid_mining_rate }
			custom_effect_tooltip = asteroid_mining_revenue_tt
		}
		else = {
			custom_effect_tooltip = asteroid_mining_not_enough_tt
		}
		hidden_effect = {
			if = {
				limit = {
					NOT = { has_global_flag = asteroid_mining_start }
				}
				set_global_flag = asteroid_mining_start
				every_country = {
					limit = {
						num_of_factories > 40
					}
					country_event = { days = 20 id = ENG_military.006 }
				}
			}
		}
	}
}

	id = devolution.3
	title = devolution.3.t
	desc = devolution.3.d

	is_triggered_only = yes

	immediate = { log = "[GetDateText]: [Root.GetName]: event devolution.3" }

	# Okay, Fine
	option = {
		name = devolution.3.a
		log = "[GetDateText]: [This.GetName]: devolution.3.a executed"
		add_to_variable = { ENG_north_irish_dependency = -1 }
		custom_effect_tooltip = ENG_decrease_dependence_1_alt_TT
		ENG_var_cap = yes
		ai_chance = {
			base = 1
		}
	}
	# No, Veto it
	option = {
		name = devolution.3.b
		log = "[GetDateText]: [This.GetName]: devolution.3.b executed"
		add_to_variable = { ENG_north_irish_agitation = 1 }
		custom_effect_tooltip = ENG_increase_agitation_1_alt_TT
		ENG_var_cap = yes
		ai_chance = {
			base = 1
		}
	}
}

# Scottish Referendum
country_event = {
	id = devolution.5
	title = devolution.5.t
	desc = devolution.5.d

	is_triggered_only = yes

	fire_only_once = yes

	# Okay, Fine
	option = {
		name = devolution.5.a
		log = "[GetDateText]: [This.GetName]: devolution.5.a executed"
		activate_mission = ENG_scottish_referendum
		ai_chance = {
			base = 1
		}
	}
	# No, Veto it
	option = {
		name = devolution.5.b
		log = "[GetDateText]: [This.GetName]: devolution.5.b executed"
		set_country_flag = ENG_scottish_rejected
		add_stability = -0.10
		every_owned_state = {
			limit = {
				is_core_of = SCO
			}
			add_dynamic_modifier = { modifier = ENG_resistance_against_westminster }
		}
		# Set a state modifier that'll rape productivity in Scotland & a decision to release them if player is fed up
		# Alternatively, if they don't want to do allat, player can take a major temporary debuff to reintegrate Scotland
		ai_chance = {
			base = 1
		}
	}
}
