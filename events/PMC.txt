news_event = {
	id = pmc_events.7
	title = pmc_events.7.t
	desc = pmc_events.7.d
	picture = GFX_trade_agreement # TODO: Need a better event picture here
	is_triggered_only = yes
	major = yes
	trigger = {
		NOT = { USA = { has_country_flag = collapsed_nation } }
		NOT = { has_global_flag = constellis_pmc_formed }
	}

	option = {
		name = pmc_events.7.a
		log = "[GetDateText]: [This.GetName]: pmc_events.7.a executed"
		trigger = { original_tag = USA }
		set_global_flag = constellis_pmc_formed
		hidden_effect = {
			every_country = {
				limit = {
					OR = {
						has_country_flag = pmc_recruited_blackwater_light
						has_country_flag = pmc_blackwater_mot
						has_country_flag = pmc_blackwater_mech
						has_country_flag = pmc_blackwater_heavy_mech
						has_country_flag = pmc_blackwater_special_unit
						has_country_flag = pmc_blackwater_tank
						has_country_flag = pmc_blackwater_specops
					}
				}
				if = { limit = { has_country_flag = pmc_recruited_blackwater_light } delete_light_infantry_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_mot } delete_mot_infantry_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_mech } delete_mech_infantry_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_heavy_mech } delete_heavy_mech_infantry_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_special_unit } delete_special_unit_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_tank } delete_tank_unit_blackwater = yes }
				if = { limit = { has_country_flag = pmc_blackwater_specops } delete_specops_unit_blackwater = yes }
			}
		}
	}

	option = {
		name = pmc_events.7.a
		trigger = { NOT = { original_tag = USA } }
	}
}
